28  /**
29   * 
30   * @author cvgaviao
31   * @since 0.5.0
32   */
33  public abstract class PluggableEventTracker {
34  
35      class KernelNotificationEventHandler implements EventHandler {
36          @Override
37          public void handleEvent(Event event) {
38              whenEventHappen(event);
39          }
40      }
41  
42      private final BundleContext bundleContext;
43  
44      private ServiceRegistration<EventHandler> eventHandlerServiceRegistration = null;
45  
46      private String filterStr;
47  
48      private List<String> topics;
49  
50      public PluggableEventTracker(BundleContext bundleContext) {
51          this(bundleContext, null, null);
52      }
53  
54      public PluggableEventTracker(BundleContext bundleContext, String[] topics) {
55          this(bundleContext, topics, "");
56      }
57  
58      public PluggableEventTracker(BundleContext bundleContext, String[] topics,
59              String filterStr) {
60          this.topics = Arrays.asList(topics);
61          this.filterStr = filterStr;
62          this.bundleContext = bundleContext;
63      }
64  
65      /**
66       * Change the properties of a registered EventHandler object.
67       * 
68       * @param topics
69       * @param filterStr
70       */
71      public void changeProperties(String[] topics, String filterStr) {
72          this.topics = Arrays.asList(topics);
73          this.filterStr = filterStr;
74          Dictionary<String, Object> props = new Hashtable<String, Object>();
75          props.put(EventConstants.EVENT_TOPIC, getTopics());
76          props.put(EventConstants.EVENT_FILTER, getFilterStr());
77  
78          eventHandlerServiceRegistration.setProperties(props);
79      }
80  
81      protected BundleContext getBundleContext() {
82          return bundleContext;
83      }
84  
85      protected String getFilterStr() {
86          return filterStr;
87      }
88  
89      protected List<String> getTopics() {
90          return Collections.unmodifiableList(topics);
91      }
92  
93      /**
94       * This method is called when a topic event is caught by the tracker.
95       * <p>
96       * Children classes must implement this method in order to start its own
97       * response actions.
98       * 
99       * @param event
100      */
101     protected abstract void whenEventHappen(Event event);
102 
103     /**
104      * Register an EventHandler object into the OSGi service registry with the
105      * properties set in the constructor.
106      */
107     public void registerEventHandler() {
108         Dictionary<String, Object> props = new Hashtable<String, Object>();
109         props.put(EventConstants.EVENT_TOPIC, getTopics());
110         props.put(EventConstants.EVENT_FILTER, getFilterStr());
111 
112         eventHandlerServiceRegistration = getBundleContext()
113                 .registerService(EventHandler.class,
114                         new KernelNotificationEventHandler(), props);
115     }
116 
117     /**
118      * Remove the EventHandler from the OSGi service registry.
119      */
120     public void unregisterEventHandler() {
121         eventHandlerServiceRegistration.unregister();
122     }
123 }
