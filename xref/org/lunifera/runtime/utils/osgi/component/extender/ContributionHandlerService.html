17  import org.osgi.framework.Bundle;
18  
19  /**
20   * A ContributionHandlerService is responsible to process the contribution
21   * resources coming from ContributorBundle components.
22   * <p>
23   * The extender utility will bind to one registered ContributionHandlerService
24   * instance and will delegate to it the resource processing job whenever some
25   * bundle matching the extender manifest header is added, modified or removed
26   * from the container.
27   * <p>
28   * A Contributor Bundle represents the <i>'extendee'</i> bundle that is
29   * contributing something (using resource files) to other <i>'extendible'</i>
30   * bundle through the <i>'extender'</i> bundle.
31   * 
32   * @author Cristiano Gavi√£o
33   * @since 0.0.1
34   */
35  public interface ContributionHandlerService {
36  
37      String EXTENSION_STRATEGY = "lunifera.extension.strategy";
38      String MANIFEST_HEADER_ITEM_TYPE = "lunifera.extender.contribution.item.resource.type";
39  
40      /**
41       * An extender implementation can process its contributions using different
42       * paths. The most commons are specified in
43       * {@link ContributionHandlerService}.
44       * 
45       * @return The handling strategy provided by this contribution handler.
46       */
47      ContributionHandlingStrategy getContributionHandlingStrategy();
48  
49      /**
50       * Each extender can support one type of resource.
51       * 
52       * @return The type of the header item supported this contribution handler.
53       */
54      ContributionItemResourceType getContributionItemResourceType();
55  
56      /**
57       * The name of this {@code ContributionHandlerService}.
58       * 
59       * @return
60       */
61      String getHandlerName();
62  
63      /**
64       * This method is called by the BundleTracker when a specific filtered
65       * bundle is arrived.
66       * 
67       * @param contributorBundle
68       * @param headerName
69       * @return
70       * @throws ExceptionComponentExtenderSetup
71       */
72      ContributorBundleTrackerObject whenContributorBundleActivated(
73              Bundle contributorBundle, String headerName, String headerValue)
74              throws ExceptionComponentExtenderSetup;
75  
76      /**
77       * This method is called by the BundleTracker when a specific filtered
78       * bundle is modified.
79       * 
80       * @param contributorBundleTrackerObject
81       * @throws ExceptionComponentExtenderSetup
82       */
83      void whenContributorBundleModified(
84              ContributorBundleTrackerObject contributorBundleTrackerObject)
85              throws ExceptionComponentExtenderSetup;
86  
87      /**
88       * This method is called by the BundleTracker when a specific filtered
89       * bundle is removed from container.
90       * 
91       * @param contributorBundleTrackerObject
92       * @throws ExceptionComponentExtenderSetup
93       */
94      void whenContributorBundleRemoved(
95              ContributorBundleTrackerObject contributorBundleTrackerObject)
96              throws ExceptionComponentExtenderSetup;
97  
98  }
