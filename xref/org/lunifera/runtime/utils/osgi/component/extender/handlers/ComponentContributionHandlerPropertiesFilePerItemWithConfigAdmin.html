82      protected final void createConfigurationForContributionItem(
83              ContributionItem contributionItem) {
84          Configuration configuration;
85  
86          // must check if the contribution is a factory or not
87  
88          try {
89              configuration = getConfigurationAdminService().getConfiguration(
90                      contributionItem.getPid(), null);
91              // Ensure update is called, when properties are null; otherwise
92              // configuration will not
93              // be returned when listConfigurations is called (see specification
94              // 104.15.3.7)
95              if (configuration.getProperties() == null) {
96                  configuration.update(new Hashtable<>(contributionItem
97                          .getPropertiesMap()));
98                  debug("Initialized store under PID: '{}', with this properties: {}",
99                          contributionItem.getPid(), contributionItem
100                                 .getPropertiesMap().toString());
101             } else {
102                 warn("Configuration for PID was already initialized: '{}'.",
103                         contributionItem.getPid());
104             }
105         } catch (IOException e) {
106             error("Error on setup Configuration Service", e);
107         }
108     }

116     private void createFactoryConfigurationForContributionItem(
117             ContributionItem contributionItem) {
118         Configuration configuration;
119 
120         // try to find an existing configuration
121         configuration = findFactoryConfigurationForContributionItem(contributionItem);
122         if (configuration == null) {
123             // create a new configuration
124             try {
125                 configuration = getConfigurationAdminService()
126                         .createFactoryConfiguration(
127                                 contributionItem.getFactoryPid(), null);
128             } catch (IOException e) {
129                 error("Error creating a Factory Config", e);
130             }
131         }
132 
133         try {
134             configuration.update(new Hashtable<>(contributionItem
135                     .getPropertiesMap()));
136 
137         } catch (IOException e) {
138             error("Error updating a Factory Config", e);
139         }
140         debug("Created configuration for FactoryPID: '{}' and PID: '{}', with this properties:\n{}",
141                 configuration.getFactoryPid(), configuration.getPid(),
142                 contributionItem.getPropertiesMap().toString());
143 
144     }

165     protected void deleteFactoryConfigurationForExtendedBundle(
166             String extenderBundle) {
167         Configuration[] configurations;
168         try {
169             configurations = findFactoryConfigurationForExtendedBundle(extenderBundle);
170             if (configurations != null) {
171                 for (Configuration configuration : configurations) {
172                     try {
173                         configuration.delete();
174                     } catch (IOException e) {
175                         error("Error on setup Configuration Service", e);
176                     }
177                 }
178             } else {
179                 debug("no configuration for factoryPid '{}' and extenderBundle '{}'",
180                         extenderBundle);
181             }
182         } catch (Exception e1) {
183             error("no configuration for factoryPid '{}' and extenderBundle '{}'",
184                     extenderBundle, e1);
185         }

189     private void extractPidsFromResourceFileName(
190             ContributionItem contributionItem) {
191         String separator = "/";
192         String filename;
193         String configFile = contributionItem.getResourceLocation().getFile();
194 
195         // Remove the path up to the filename.
196         int lastSeparatorIndex = configFile.lastIndexOf(separator);
197         if (lastSeparatorIndex == -1) {
198             filename = configFile;
199         } else {
200             filename = configFile.substring(lastSeparatorIndex + 1);
201         }
202 
203         // Factory (if it exists)
204         int qualifierIndex = filename.lastIndexOf('\u005f');
205         int extensionIndex = filename.lastIndexOf('\u002e');
206         if (qualifierIndex != -1) {
207             contributionItem.getPropertiesMap().put(SERVICE_FACTORYPID,
208                     filename.substring(0, qualifierIndex));
209             if (extensionIndex != -1) {
210                 String pid = filename.substring(qualifierIndex + 1,
211                         extensionIndex);
212                 if (!pid.isEmpty()) {
213                     contributionItem.getPropertiesMap().put(
214                             SERVICE_PID,
215                             filename.substring(qualifierIndex + 1,
216                                     extensionIndex));
217                 }
218             } else {
219                 contributionItem.getPropertiesMap().put(SERVICE_PID,
220                         filename.substring(qualifierIndex));
221             }
222         } else {
223             if (extensionIndex == -1) {
224                 contributionItem.getPropertiesMap().put(SERVICE_PID, filename);
225             } else {
226                 contributionItem.getPropertiesMap().put(SERVICE_PID,
227                         filename.substring(0, extensionIndex));
228             }
229         }
230     }

232     protected Configuration findConfiguration(String pid) throws IOException {
233         // As ConfigurationAdmin.getConfiguration creates the configuration if
234         // it is not yet there, we check its existence first
235         try {
236             Configuration[] configurations = getConfigurationAdminService()
237                     .listConfigurations("(service.pid=" + pid + ")");
238             if (configurations != null && configurations.length > 0) {
239                 return configurations[0];
240             }
241         } catch (InvalidSyntaxException e) {
242             error("Error on findConfiguration", e);
243         }
244 
245         return null;
246     }

248     protected final Configuration findFactoryConfigurationForContributionItem(
249             ContributionItem contributionItem) {
250         try {
251             Configuration[] configurations = getConfigurationAdminService()
252                     .listConfigurations(contributionItem.getFilter());
253             if (configurations != null && configurations.length > 0) {
254                 return configurations[0];
255             }
256         } catch (Exception e) {
257             error("Error on findFactoryConfigurationByContributionItem", e);
258         }
259         return null;
260     }

298     @Override
299     public ContributorBundleTrackerObject whenContributorBundleActivated(
300             Bundle contributorBundle, String headerName, String headerValue)
301             throws ExceptionComponentExtenderSetup {
302         ContributorBundleTrackerObject result = super
303                 .whenContributorBundleActivated(contributorBundle, headerName,
304                         headerValue);
305         // process every ContributionItem created and create a proper
306         // Configuration for them
307         for (ContributionItem contributionItem : result
308                 .getContributionItemList()) {
309 
310             extractPidsFromResourceFileName(contributionItem);
311 
312             if (contributionItem.getFactoryPid() != null) {
313                 createFactoryConfigurationForContributionItem(contributionItem);
314             } else {
315                 createConfigurationForContributionItem(contributionItem);
316             }
317         }
318         return result;
319     }
