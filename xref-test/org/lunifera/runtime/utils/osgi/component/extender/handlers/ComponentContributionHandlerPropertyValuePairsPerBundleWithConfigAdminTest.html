37      static class ConcreteHandler extends
38              AbstractComponentFactoryContributionHandlerPropertiesFilePerItem {
39  
40          public ConcreteHandler() {
41              super();
42          }
43  
44          public ConcreteHandler(
45                  ContributionItemResourceType contributionItemResourceType,
46                  ContributionHandlingStrategy extensionHandlingStrategy) {
47              super(contributionItemResourceType, extensionHandlingStrategy);
48          }
49  
50      }
51  
52      ConcreteHandler handler;
53  
54      @Before
55      public void setup() {
56          handler = new ConcreteHandler();
57      }
58  
59      @SuppressWarnings("unchecked")
60      // @Test
61      public void testOnContributorBundleAddition_PROPERTY_VALUE_PAIR_PER_BUNDLE()
62              throws Exception {
63          Bundle mockBundle = mock(Bundle.class);
64          ContributorBundleTrackerObject contributorBundleTrackerObject;
65          ConcreteHandler spyHandler = spy(new ConcreteHandler(
66                  ContributionItemResourceType.PROPERTY_VALUE_PAIRS,
67                  ContributionHandlingStrategy.PER_BUNDLE));
68  
69          when(mockBundle.getHeaders()).thenReturn(mock(Dictionary.class));
70          when(mockBundle.getHeaders().get("My_Header")).thenReturn(
71                  "key0=value0,key1=value1,key2=value2");
72          when(mockBundle.getSymbolicName()).thenReturn("my.bundle");
73          when(mockBundle.getVersion()).thenReturn(new Version("1.0.0"));
74  
75          contributorBundleTrackerObject = spyHandler
76                  .whenContributorBundleActivated(mockBundle, "My_Header",
77                          "pro1=value1");
78  
79          assertNotNull(contributorBundleTrackerObject);
80          assertThat(contributorBundleTrackerObject.getContributionItemList(),
81                  hasSize(1));
82      }
